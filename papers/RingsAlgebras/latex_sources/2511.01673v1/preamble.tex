%   Macros for finite sums (\sumprimed) and finite products (\prodprimed)

\def\sumprime{\mathop{\sum\nolimits'}}
\def\prodprime{\mathop{\prod\nolimits'}}

\def\sumpr#1{\sumprime\limits_{#1}}

\def\dsumprime#1#2{\sum_{#1}^{#2}\!\vphantom{\bigg|}^{\textstyle{}'}}
\def\dprodprime#1#2{\prod_{#1}^{#2}\!\vphantom{\bigg|}^{\textstyle{}'}}
\def\tsumprime#1#2{{\sum'}_{\!\!#1}^{#2}}
\def\tprodprime#1#2{{\prod'}_{\!\!\!#1}^{#2}}

\def\sumprimed#1#2{\mathchoice{\dsumprime{#1}{#2}}{\tsumprime{#1}{#2}}
{\scriptstyle \tsumprime{#1}{#2}}{\scriptscriptstyle \tsumprime{#1}{#2}}}
\def\prodprimed#1#2{\mathchoice{\dprodprime{#1}{#2}}{\tprodprime{#1}{#2}}
{\scriptstyle \tprodprime{#1}{#2}}{\scriptscriptstyle \tprodprime{#1}{#2}}}

\def\qed{{\hskip0pt\unskip\unskip\nobreak\hfil\penalty50
          \hskip1em\hbox{}\nobreak\hfil
           {$\square$}
          \parfillskip=0pt\finalhyphendemerits=0
          \par}}

               {\noindent{\textit{Proof}.}\;}
               {\qed}
